  // Trigonometric
  'sin': math.sin,
  'cos': math.cos,
  'tan': math.tan,
  'asin': math.asin,
  'acos': math.acos,
  'atan': math.atan,
  'atan2': math.atan2,

  // Hyperbolic
  'sinh': _sinh,
  'cosh': _cosh,
  'tanh': _tanh,

  // Power and roots

  'pow': math.pow,
  'exp': math.exp,

  'max': math.max,

  // Constants
  'pi': math.pi,
  'e': math.e,
};
